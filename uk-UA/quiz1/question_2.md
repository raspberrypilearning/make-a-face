def draw():

  #Обличчя
  stroke(0) #Чорний
  fill(255) #Білий
  ellipse(200, 200, 200, 190)
  no_stroke()
  
  #Очі
  fill(0, 255, 0) #Зелений
  ellipse(160, 180, 60, 60)
  ellipse(240, 180, 60, 60)
  fill(0) #Чорний
  ellipse(160, 180, 30, 30)
  ellipse(240, 180, 30, 30)
  
run()
